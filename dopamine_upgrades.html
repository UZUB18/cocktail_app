interface Substitute {
  name: string;
  compatibility: number; // 0.0 - 1.0
  notes: string;
}

function getSubstitutes(missing: string): Substitute[] {
  const dataset: Record = {
    "blanco tequila": [
      { name: "Mezcal", compatibility: 0.91, notes: "Adds smoky qualities." },
      { name: "Sotol", compatibility: 0.82, notes: "Earthy and herbal." }
    ],
    "bourbon": [
      { name: "Rye Whiskey", compatibility: 0.85, notes: "Spicier flavor profile." },
      { name: "Cognac", compatibility: 0.80, notes: "Fruity and sweet profile." }
    ]
  };
  return dataset[missing.toLowerCase()] || [];
}

function getRadarPoints(profile: FlavorProfile) {
  const center = 50;
  const scale = 40; // Max radius
  // Convert low/med/high to coordinate values
  const valMap = { low: 0.25, medium: 0.5, high: 0.85 };
  
  const sweetY = center - valMap[profile.sweetness] * scale;
  const sourX = center - valMap[profile.acidity] * scale;
  const bitterY = center + valMap[profile.bitterness] * scale;
  const boozeX = center + valMap[profile.booziness] * scale;
  
  return `${center},${sweetY} ${boozeX},${center} ${center},${bitterY} ${sourX},${center}`;
}

Batch Scaling Calculation logic 
function calculateBatch(recipe: Recipe, servings: number) {
  const dilutionModifier = servings > 10 ? 0.85 : 1.0;
  return recipe.ingredients.map(ing => {
    let amt = ing.amount * servings;
    if (ing.name === "water" || ing.name === "ice") {
      amt = amt * dilutionModifier; // scale dilution safely
    }
    return { ...ing, amount: amt };
  });
}

const SpeechRecognition = window.SpeechRecognition || window.webkitSpeechRecognition;
if (SpeechRecognition) {
  const recognition = new SpeechRecognition();
  recognition.continuous = true;
  recognition.onresult = (event) => {
    const transcript = event.results[event.results.length - 1][0].transcript.trim().toLowerCase();
    if (transcript.includes('next')) {
      goToNextStep();
    } else if (transcript.includes('back')) {
      goToPrevStep();
    }
  };
  recognition.start();
}
